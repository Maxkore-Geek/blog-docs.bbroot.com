🤖 自动备份脚本

在博客服务器上设置定时任务：

#!/bin/bash
# 博客自动备份脚本
# 每天凌晨 3 点执行

BACKUP_DIR="/e/blog-docs.bbroot.com/backups/posts"
DATE=$(date +%Y%m%d)

# 备份文章
tar -czf "$BACKUP_DIR/posts_$DATE.tar.gz" /e/MyBlog/source/_posts/

# 备份配置
cp /e/MyBlog/_config.yml "/e/blog-docs.bbroot.com/backups/config/config_$DATE.yml"

# 备份数据库
cp /e/MyBlog/db.json "/e/blog-docs.bbroot.com/backups/db/db_$DATE.json"

# 提交到 Git
cd /e/blog-docs.bbroot.com
git add .
git commit -m "Daily backup $DATE"
git push

echo "✅ 备份完成 $DATE"
